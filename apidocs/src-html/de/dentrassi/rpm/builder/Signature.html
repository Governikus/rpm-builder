011package de.dentrassi.rpm.builder;
012
013import java.io.File;
014
015public class Signature
016{
017    private String keyId;
018
019    private File keyringFile;
020
021    private String passphrase;
022
023    private String hashAlgorithm = "SHA512";
024
025    private boolean skip;
026
027    public void setKeyId ( final String keyId )
028    {
029        this.keyId = keyId;
030    }
031
032    public String getKeyId ()
033    {
034        return this.keyId;
035    }
036
037    public void setKeyringFile ( final File keyringFile )
038    {
039        this.keyringFile = keyringFile;
040    }
041
042    public File getKeyringFile ()
043    {
044        return this.keyringFile;
045    }
046
047    public void setPassphrase ( final String passphrase )
048    {
049        this.passphrase = passphrase;
050    }
051
052    public String getPassphrase ()
053    {
054        return this.passphrase;
055    }
056
057    public void setHashAlgorithm ( final String hashAlgorithm )
058    {
059        this.hashAlgorithm = hashAlgorithm;
060    }
061
062    public String getHashAlgorithm ()
063    {
064        return this.hashAlgorithm;
065    }
066
067    public void setSkip ( final boolean skip )
068    {
069        this.skip = skip;
070    }
071
072    public boolean isSkip ()
073    {
074        return this.skip;
075    }
076}
