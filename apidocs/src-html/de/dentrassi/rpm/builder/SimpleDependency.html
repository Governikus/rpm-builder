011package de.dentrassi.rpm.builder;
012
013import java.util.Arrays;
014import java.util.HashMap;
015import java.util.HashSet;
016import java.util.Map;
017import java.util.Set;
018
019import org.eclipse.packagedrone.utils.rpm.deps.RpmDependencyFlags;
020
021public class SimpleDependency
022{
023    protected static final Map<String, RpmDependencyFlags[]> OP_MAP;
024
025    static
026    {
027        OP_MAP = new HashMap<> ();
028
029        OP_MAP.put ( "=", new RpmDependencyFlags[] { RpmDependencyFlags.EQUAL } );
030        OP_MAP.put ( ">=", new RpmDependencyFlags[] { RpmDependencyFlags.EQUAL, RpmDependencyFlags.GREATER } );
031        OP_MAP.put ( "<=", new RpmDependencyFlags[] { RpmDependencyFlags.EQUAL, RpmDependencyFlags.LESS } );
032        OP_MAP.put ( ">", new RpmDependencyFlags[] { RpmDependencyFlags.GREATER } );
033        OP_MAP.put ( "<", new RpmDependencyFlags[] { RpmDependencyFlags.LESS } );
034
035        OP_MAP.put ( "eq", new RpmDependencyFlags[] { RpmDependencyFlags.EQUAL } );
036        OP_MAP.put ( "ge", new RpmDependencyFlags[] { RpmDependencyFlags.EQUAL, RpmDependencyFlags.GREATER } );
037        OP_MAP.put ( "le", new RpmDependencyFlags[] { RpmDependencyFlags.EQUAL, RpmDependencyFlags.LESS } );
038        OP_MAP.put ( "gt", new RpmDependencyFlags[] { RpmDependencyFlags.GREATER } );
039        OP_MAP.put ( "lt", new RpmDependencyFlags[] { RpmDependencyFlags.LESS } );
040    }
041
042    protected String name;
043
044    protected String version;
045
046    protected Set<RpmDependencyFlags> flags = new HashSet<> ();
047
048    public void set ( final String string )
049    {
050        final String[] toks = string.split ( "\\s+" );
051        if ( toks.length == 1 )
052        {
053            setAll ( toks[0], null );
054        }
055        else if ( toks.length == 3 )
056        {
057            final RpmDependencyFlags[] flags = OP_MAP.get ( toks[1] );
058            if ( flags == null )
059            {
060                throw new IllegalArgumentException ( String.format ( "Operator '%s' is unknown", toks[1] ) );
061            }
062            setAll ( toks[0], toks[2], flags );
063        }
064        else
065        {
066            throw new IllegalArgumentException ( String.format ( "Invalid short format: '%s'", string ) );
067        }
068    }
069
070    private void setAll ( final String name, final String version, final RpmDependencyFlags... flags )
071    {
072        this.name = name;
073        this.version = version;
074        this.flags.clear ();
075        if ( flags != null )
076        {
077            this.flags.addAll ( Arrays.asList ( flags ) );
078        }
079    }
080
081    public String getName ()
082    {
083        return this.name;
084    }
085
086    public String getVersion ()
087    {
088        return this.version;
089    }
090
091    public Set<RpmDependencyFlags> getFlags ()
092    {
093        return this.flags;
094    }
095}
