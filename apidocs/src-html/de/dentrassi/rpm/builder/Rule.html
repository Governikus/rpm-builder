011package de.dentrassi.rpm.builder;
012
013import org.eclipse.packagedrone.utils.rpm.build.PayloadEntryType;
014import org.slf4j.LoggerFactory;
015
016public class Rule extends EntryDetails
017{
018    private static final org.slf4j.Logger logger = LoggerFactory.getLogger ( Rule.class );
019
020    private When when;
021
022    private boolean last = false;
023
024    public void setWhen ( final When when )
025    {
026        this.when = when;
027    }
028
029    public When getWhen ()
030    {
031        return this.when;
032    }
033
034    public void setLast ( final boolean last )
035    {
036        this.last = last;
037    }
038
039    public boolean isLast ()
040    {
041        return this.last;
042    }
043
044    public boolean matches ( final Object object, final PayloadEntryType type, final String targetName )
045    {
046        if ( this.when == null )
047        {
048            logger.debug ( "No 'when'" );
049            return true;
050        }
051        else
052        {
053            logger.debug ( "Matching 'when': {}", this.when );
054            return this.when.matches ( object, type, targetName );
055        }
056    }
057
058    @Override
059    public String toString ()
060    {
061        return String.format ( "[Rule - when: %s, then: %s]", this.when, super.toString () );
062    }
063}
