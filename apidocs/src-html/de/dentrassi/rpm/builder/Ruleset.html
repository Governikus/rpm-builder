011package de.dentrassi.rpm.builder;
012
013import java.util.LinkedList;
014import java.util.List;
015
016public class Ruleset
017{
018    private String id;
019
020    private List<Rule> rules = new LinkedList<> ();
021
022    private String defaultRuleset;
023
024    public void setId ( final String id )
025    {
026        this.id = id;
027    }
028
029    public String getId ()
030    {
031        return this.id;
032    }
033
034    public void setRules ( final List<Rule> rules )
035    {
036        this.rules = rules;
037    }
038
039    public List<Rule> getRules ()
040    {
041        return this.rules;
042    }
043
044    public void setDefaultRuleset ( final String defaultRuleset )
045    {
046        this.defaultRuleset = defaultRuleset;
047    }
048
049    public String getDefaultRuleset ()
050    {
051        return this.defaultRuleset;
052    }
053
054    public void validate ()
055    {
056        for ( final Rule rule : this.rules )
057        {
058            rule.validate ();
059        }
060    }
061}
