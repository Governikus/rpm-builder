012package de.dentrassi.rpm.builder;
013
014import org.apache.maven.plugin.logging.Log;
015
016public class Logger
017{
018    private final Log log;
019
020    public Logger ( final Log log )
021    {
022        this.log = log;
023    }
024
025    public void debug ( final String format, final Object... values )
026    {
027        this.log.debug ( String.format ( format, values ) );
028    }
029
030    public void info ( final String format, final Object... values )
031    {
032        this.log.info ( String.format ( format, values ) );
033    }
034
035    public void warn ( final String format, final Object... values )
036    {
037        this.log.warn ( String.format ( format, values ) );
038    }
039}
