061/**
062 * Build a YUM repository
063 *
064 * @author ctron
065 */
066@Mojo ( name = "yum-repository",
067        defaultPhase = LifecyclePhase.PACKAGE,
068        requiresProject = true,
069        requiresDependencyResolution = ResolutionScope.COMPILE,
070        requiresDependencyCollection = ResolutionScope.COMPILE,
071        threadSafe = false )
072public class YumMojo extends AbstractMojo
073{
074    /**
075     * The maven project
076     */
077    @Parameter ( property = "project", readonly = true, required = true )
078    protected MavenProject project;
079
080    @Parameter ( defaultValue = "${session}", readonly = true, required = true )
081    private MavenSession session;
082
083    @Component
084    private MavenProjectHelper projectHelper;
085
086    /**
087     * The location of the output directory.
088     * <p>
089     * This directory will be created if it does not exists.
090     * </p>
091     */
092    @Parameter ( property = "yum.repository.output", defaultValue = "${project.build.directory}/yum" )
093    private File outputDirectory;
094
095    /**
096     * A list of files to be added
097     * <p>
098     * All listed files will be added to the final repository.
099     * </p>
100     */
101    @Parameter
102    private List<File> files;
103
104    /**
105     * A list of directories
106     * <p>
107     * These directories will be scanned recursively for files with the
108     * extension <code>.rpm</code>. All matching files will be added to the
109     * final repository.
110     * </p>
111     */
112    @Parameter
113    private List<File> directories;
114
115    private File packagesPath;
116
117    /**
118     * An optional signature descriptor for GPP signing the final YUM repository
119     * <p>
120     * Also see <a href="signing.html">signing</a>
121     * </p>
122     */
123    @Parameter ( property = "rpm.signature" )
124    private Signature signature;
125
126    /**
127     * Disable all repository signing
128     */
129    @Parameter ( property = "rpm.skipSigning", defaultValue = "false" )
130    private boolean skipSigning = false;
131
132    /**
133     * Disable the use of RPMs from maven dependency artifacts
134     */
135    @Parameter ( property = "rpm.skipDependencies", defaultValue = "false" )
136    private boolean skipDependencies = false;
137
138    /**
139     * Disable the mojo altogether.
140     *
141     * @since 1.1.1
142     */
143    @Parameter ( property = "yum.skip", defaultValue = "false" )
144    private boolean skip;
145
146    public void setSkip ( final boolean skip )
147    {
148        this.skip = skip;
149    }
150
151    private Logger logger;
152
153    @Override
154    public void execute () throws MojoExecutionException, MojoFailureException
155    {
156        this.logger = new Logger ( getLog () );
157
158        if ( this.skip )
159        {
160            this.logger.debug ( "Skipping execution" );
161            return;
162        }
163
164        try
165        {
166            final Builder builder = new RepositoryCreator.Builder ();
167            builder.setTarget ( new FileSystemSpoolOutTarget ( this.outputDirectory.toPath () ) );
168
169            if ( !this.skipSigning )
170            {
171                final PGPPrivateKey privateKey = SigningHelper.loadKey ( this.signature, this.logger );
172                if ( privateKey != null )
173                {
174                    final int digestAlgorithm = HashAlgorithm.from ( this.signature.getHashAlgorithm () ).getValue ();
175                    builder.setSigning ( output -> new SigningStream ( output, privateKey, digestAlgorithm, false, "RPM builder Mojo - de.dentrassi.maven:rpm" ) );
176                }
177            }
178
179            final RepositoryCreator creator = builder.build ();
180
181            this.packagesPath = new File ( this.outputDirectory, "packages" );
182            Files.createDirectories ( this.packagesPath.toPath () );
183
184            final Collection<Path> paths = Lists.newArrayList ();
185
186            if ( !this.skipDependencies )
187            {
188                final Set<Artifact> deps = this.project.getArtifacts ();
189                if ( deps != null )
190                {
191                    paths.addAll ( deps.stream ()//
192                            .filter ( d -> d.getType ().equalsIgnoreCase ( "rpm" ) )//
193                            .map ( d -> d.getFile ().toPath () )//
194                            .collect ( Collectors.toList () ) );
195                }
196            }
197            else
198            {
199                this.logger.debug ( "Skipped RPM artifacts from maven dependencies" );
200            }
201
202            if ( this.files != null )
203            {
204                paths.addAll ( this.files.stream ().map ( f -> f.toPath () ).collect ( Collectors.toList () ) );
205            }
206            if ( this.directories != null )
207            {
208                for ( final File dir : this.directories )
209                {
210                    Files.walkFileTree ( dir.toPath (), new SimpleFileVisitor<Path> () {
211                        @Override
212                        public FileVisitResult visitFile ( final Path file, final BasicFileAttributes attrs ) throws IOException
213                        {
214                            if ( file.getFileName ().toString ().toLowerCase ().endsWith ( ".rpm" ) )
215                            {
216                                paths.add ( file );
217                            }
218                            return FileVisitResult.CONTINUE;
219                        }
220                    } );
221                }
222            }
223
224            addPackageList ( creator, paths );
225        }
226        catch ( final IOException e )
227        {
228            throw new MojoExecutionException ( "Failed to write repository", e );
229        }
230    }
231
232    private void addPackageList ( final RepositoryCreator creator, final Collection<Path> paths )
233    {
234        try
235        {
236            creator.process ( context -> {
237                for ( final Path p : paths )
238                {
239                    addSinglePackage ( p, context );
240                }
241                getLog ().info ( String.format ( "Added %s packages to the repository", paths.size () ) );
242            } );
243        }
244        catch ( final IOException e )
245        {
246            throw new RuntimeException ( e );
247        }
248    }
249
250    private void addSinglePackage ( final Path path, final Context context ) throws IOException
251    {
252        final String checksum = makeChecksum ( path );
253        final String fileName = path.getFileName ().toString ();
254        final String location = "packages/" + fileName;
255        final FileInformation fileInformation = new FileInformation ( Files.getLastModifiedTime ( path ).toInstant (), Files.size ( path ), location );
256
257        final RpmInformation rpmInformation;
258        try ( RpmInputStream ris = new RpmInputStream ( Files.newInputStream ( path ) ) )
259        {
260            rpmInformation = RpmInformations.makeInformation ( ris );
261        }
262
263        context.addPackage ( fileInformation, rpmInformation, singletonMap ( SHA256, checksum ), SHA256 );
264
265        Files.copy ( path, this.packagesPath.toPath ().resolve ( fileName ), StandardCopyOption.COPY_ATTRIBUTES );
266    }
267
268    private String makeChecksum ( final Path path ) throws IOException
269    {
270        try ( InputStream is = Files.newInputStream ( path ) )
271        {
272            return DigestUtils.sha256Hex ( is ).toLowerCase ();
273        }
274    }
