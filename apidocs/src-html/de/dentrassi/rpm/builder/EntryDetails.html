021public class EntryDetails
022{
023    private Short mode;
024
025    private Boolean configuration;
026
027    private Boolean documentation;

032

034

038

040

044

155    public void validate ()
156    {
157    }
158
159    public boolean apply ( final FileInformation info )
160    {
161        boolean didApply = false;
162
163        if ( this.configuration != null )
164        {
165            setFlag ( info, FileFlags.CONFIGURATION );
166            didApply = true;
167        }
168        if ( this.documentation != null )
169        {
170            setFlag ( info, FileFlags.DOC );
171            didApply = true;
172        }
173        if ( this.license != null )
174        {
175            setFlag ( info, FileFlags.LICENSE );
176            didApply = true;
177        }
178        if ( this.readme != null )
179        {
180            setFlag ( info, FileFlags.README );
181            didApply = true;
182        }
183        if ( this.ghost != null )
184        {
185            setFlag ( info, FileFlags.GHOST );
186            didApply = true;
187        }
188        if ( this.missingOk != null )
189        {
190            setFlag ( info, FileFlags.MISSINGOK );
191            didApply = true;
192        }
193        if ( this.noreplace != null )
194        {
195            setFlag ( info, FileFlags.NOREPLACE );
196            didApply = true;
197        }
198        if ( this.user != null && !this.user.isEmpty () )
199        {
200            info.setUser ( this.user );
201            didApply = true;
202        }
203        if ( this.group != null && !this.group.isEmpty () )
204        {
205            info.setGroup ( this.group );
206            didApply = true;
207        }
208        if ( this.mode != null )
209        {
210            info.setMode ( this.mode );
211            didApply = true;
212        }
213        return didApply;
214    }
215
216    private void setFlag ( final FileInformation info, final FileFlags flag )
217    {
218        final Set<FileFlags> flags = info.getFileFlags ();
219        if ( flags == null )
220        {
221            info.setFileFlags ( EnumSet.of ( flag ) );
222        }
223        else
224        {
225            flags.add ( flag );
226        }
227    }
