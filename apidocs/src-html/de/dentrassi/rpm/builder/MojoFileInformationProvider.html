012package de.dentrassi.rpm.builder;
013
014import java.io.IOException;
015import java.util.Objects;
016import java.util.function.Consumer;
017
018import org.eclipse.packagedrone.utils.rpm.build.BuilderContext;
019import org.eclipse.packagedrone.utils.rpm.build.FileInformation;
020import org.eclipse.packagedrone.utils.rpm.build.FileInformationProvider;
021import org.eclipse.packagedrone.utils.rpm.build.PayloadEntryType;
022
023public class MojoFileInformationProvider implements FileInformationProvider<Object>
024{
025    private final RulesetEvaluator rulesetEval;
026
027    private final Consumer<String> logger;
028
029    private final String ruleId;
030
031    private final PackageEntry entry;
032
033    public MojoFileInformationProvider ( final RulesetEvaluator rulesetEval, final String ruleId, final PackageEntry entry, final Consumer<String> logger )
034    {
035        this.rulesetEval = Objects.requireNonNull ( rulesetEval );
036        this.ruleId = ruleId;
037        this.entry = entry;
038        this.logger = logger != null ? logger : ( s ) -> {
039        };
040    }
041
042    @Override
043    public FileInformation provide ( final String targetName, final Object object, final PayloadEntryType type ) throws IOException
044    {
045        final FileInformation result = provideByRule ( targetName, object, type );
046
047        if ( result == null )
048        {
049            throw new IllegalStateException ( "Unable to provide file information" );
050        }
051
052        if ( this.entry != null )
053        {
054            if ( this.entry.apply ( result ) )
055            {
056                this.logger.accept ( String.format ( "local override = %s", result ) );
057            }
058        }
059
060        return result;
061    }
062
063    private FileInformation provideByRule ( final String targetName, final Object object, final PayloadEntryType type ) throws IOException
064    {
065        final FileInformation result = BuilderContext.defaultProvider ().provide ( targetName, object, type );
066
067        if ( this.ruleId != null && !this.ruleId.isEmpty () )
068        {
069            this.logger.accept ( String.format ( "run ruleset: '%s'", this.ruleId ) );
070            this.rulesetEval.eval ( this.ruleId, object, type, targetName, result );
071        }
072
073        this.logger.accept ( String.format ( "fileInformation = %s", result ) );
074
075        return result;
076    }
077}
