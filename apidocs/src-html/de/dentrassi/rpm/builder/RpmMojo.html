642    @Override
643    public void execute () throws MojoExecutionException, MojoFailureException
644    {
645        this.logger = new Logger ( getLog () );
646
647        if ( this.skip )
648        {
649            this.logger.debug ( "Skipping execution" );
650            return;
651        }
652
653        this.eval = new RulesetEvaluator ( this.rulesets );
654
655        final Path targetDir;
656
657        if ( this.targetDir != null )
658        {
659            targetDir = this.targetDir.toPath ();
660        }
661        else
662        {
663            targetDir = Paths.get ( this.project.getBuild ().getDirectory () );
664        }
665
666        if ( !Files.exists ( targetDir ) )
667        {
668            try
669            {
670                Files.createDirectories ( targetDir );
671            }
672            catch ( final FileAlreadyExistsException e )
673            {
674                // silently ignore
675            }
676            catch ( final IOException ioe )
677            {
678                this.logger.debug ( "Unable to create target directory {}", targetDir );
679                throw new MojoExecutionException ( "RPM build failed.", ioe );
680
681            }
682        }
683
684        final Path targetFile = makeTargetFile ( targetDir );
685
686        this.logger.debug ( "Max supported RPM version: %s", this.maximumSupportedRpmVersion );
687
688        this.logger.info ( "Writing to target to: %s", targetFile );
689        this.logger.debug ( "Default script interpreter: %s", this.defaultScriptInterpreter );
690        this.logger.debug ( "Default ruleset: %s", this.defaultRuleset );
691
692        final String packageName = makePackageName ();
693        final RpmVersion version = makeVersion ();
694
695        this.logger.info ( "RPM base information - name: %s, version: %s, arch: %s", packageName, version, this.architecture );
696
697        testLeadFlags ();
698
699        try ( final RpmBuilder builder = new RpmBuilder ( packageName, version, this.architecture, targetFile ) )
700        {
701            this.logger.info ( "Writing target file: %s", builder.getTargetFile () );
702
703            if ( this.leadOverrideArchitecture != null )
704            {
705                this.logger.info ( "Override RPM lead architecture: %s", this.leadOverrideArchitecture );
706                builder.setLeadOverrideArchitecture ( this.leadOverrideArchitecture );
707            }
708            if ( this.leadOverrideOperatingSystem != null )
709            {
710                this.logger.info ( "Override RPM lead operating system: %s", this.leadOverrideOperatingSystem );
711                builder.setLeadOverrideOperatingSystem ( this.leadOverrideOperatingSystem );
712            }
713
714            fillPackageInformation ( builder );
715            fillScripts ( builder );
716            fillDependencies ( builder );
717            fillPayload ( builder );
718            fillPrefixes ( builder );
719
720            // add signer
721
722            if ( !this.skipSigning && this.signature != null )
723            {
724                final SignatureProcessor signer = makeRsaSigner ( this.signature );
725                if ( signer != null )
726                {
727                    builder.addSignatureProcessor ( signer );
728                }
729            }
730
731            // finally build the file
732
733            builder.build ();
734
735            // version check
736
737            checkVersion ( builder );
738
739            // attach when necessary
740
741            if ( this.attach )
742            {
743                this.projectHelper.attachArtifact ( this.project, "rpm", this.classifier, builder.getTargetFile ().toFile () );
744            }
745        }
746        catch ( final IOException e )
747        {
748            throw new MojoExecutionException ( "Failed to write RPM", e );
749        }
750    }
751
752    private Path makeTargetFile ( final Path targetDir )
753    {
754        String outputFileName = this.outputFileName;
755
756        if ( outputFileName == null || outputFileName.isEmpty () )
757        {
758            switch ( this.naming.getDefaultFormat () )
759            {
760                case LEGACY:
761                    outputFileName = RpmFileNameProvider.LEGACY_FILENAME_PROVIDER.getRpmFileName ( makePackageName (), makeVersion (), this.architecture );
762                    break;
763                default:
764                    outputFileName = RpmFileNameProvider.DEFAULT_FILENAME_PROVIDER.getRpmFileName ( makePackageName (), makeVersion (), this.architecture );
765                    break;
766            }
767            this.logger.debug ( "Using generated file name - %s", outputFileName, outputFileName );
768        }
769
770        final Path targetFile = targetDir.resolve ( outputFileName );;
771        this.logger.debug ( "Resolved output file name - fileName: %s, fullName: %s", this.outputFileName, targetFile );
772        return targetFile;
773    }

859    private <T extends SimpleDependency> void addAllDependencies ( final String depName, final List<T> deps, final DependencyAdder adder, final Consumer<T> validator, final Consumer<Set<RpmDependencyFlags>> flagsCustomizer )
860    {
861        if ( deps == null )
862        {
863            return;
864        }
865
866        for ( final T dep : deps )
867        {
868            validator.accept ( dep );
869
870            final String name = dep.getName ();
871            final String version = dep.getVersion ();
872            final Set<RpmDependencyFlags> flags = dep.getFlags ();
873
874            if ( flagsCustomizer != null )
875            {
876                flagsCustomizer.accept ( flags );
877            }
878
879            this.logger.info ( "Adding dependency [%s]: name = %s, version = %s, flags = %s", depName, name, version, flags );
880
881            adder.add ( name, version, flags.toArray ( new RpmDependencyFlags[0] ) );
882        }

893    private void setScript ( final String scriptName, final Script script, final ScriptSetter setter ) throws IOException
894    {
895        if ( script == null )
896        {
897            return;
898        }
899
900        final String scriptContent = script.makeScriptContent ();
901
902        if ( Strings.isNullOrEmpty ( scriptContent ) )
903        {
904            return;
905        }
906
907        String interpreter = script.getInterpreter ();
908        this.logger.debug ( "[script %s:]: explicit interpreter: %s", scriptName, interpreter );
909
910        if ( Strings.isNullOrEmpty ( interpreter ) )
911        {
912            interpreter = detectInterpreter ( scriptContent );
913            this.logger.debug ( "[script %s:]: detected interpreter: %s", scriptName, interpreter );
914        }
915        if ( Strings.isNullOrEmpty ( interpreter ) )
916        {
917            interpreter = this.defaultScriptInterpreter;
918            this.logger.debug ( "[script %s:]: default interpreter: %s", scriptName, interpreter );
919        }
920        this.logger.info ( "[script %s]: Using script interpreter: %s", scriptName, interpreter );
921        this.logger.debug ( "[script %s]: %s", scriptName, scriptContent );
922
923        setter.accept ( interpreter, scriptContent );
924    }
925
926    private String detectInterpreter ( final String scriptContent ) throws IOException
927    {
928        final String firstLine = CharSource.wrap ( scriptContent ).readFirstLine ();
929        if ( Strings.isNullOrEmpty ( firstLine ) )
930        {
931            return null;
932        }
933
934        if ( firstLine.startsWith ( "#!" ) && firstLine.length () > 2 )
935        {
936            return firstLine.substring ( 2 );
937        }
938
939        return null;
940    }

943    {
944        if ( this.entries == null )
945        {

947        }

1028    private void fillFromEntryCollect ( final BuilderContext ctx, final PackageEntry entry ) throws IOException
1029    {
1030        this.logger.debug ( "    as collector:" );
1031
1032        final Collector collector = entry.getCollect ();
1033
1034        this.logger.debug ( "      - configuration: %s", collector );
1035
1036        final String padding = "          ";
1037
1038        final Path from = collector.getFrom ().toPath ();
1039        final String targetPrefix = entry.getName ().endsWith ( "/" ) ? entry.getName () : entry.getName () + "/";
1040
1041        this.logger.debug ( "      - files:" );
1042
1043        final MojoFileInformationProvider provider = makeProvider ( entry, "            - " );
1044
1045        final DirectoryScanner scanner = new DirectoryScanner ();
1046        scanner.setBasedir ( from.toFile () );
1047        scanner.setCaseSensitive ( true );
1048        scanner.setFollowSymlinks ( true );
1049        scanner.setIncludes ( collector.getIncludes () );
1050        scanner.setExcludes ( collector.getExcludes () );
1051        scanner.scan ();
1052
1053        if ( collector.isDirectories () )
1054        {
1055            for ( final String directory : scanner.getIncludedDirectories () )
1056            {
1057                final Path dir = from.resolve ( directory );
1058                if ( dir.equals ( from ) )
1059                {
1060                    continue;
1061                }
1062
1063                RpmMojo.this.logger.debug ( "%s%s (dir)", padding, dir );
1064                final Path relative = from.relativize ( dir );
1065                final String targetName = makeUnix ( targetPrefix + relative.toString () );
1066                RpmMojo.this.logger.debug ( "%s  - target: %s", padding, targetName );
1067                ctx.addDirectory ( targetName, provider );
1068            }
1069        }
1070
1071        for ( final String relative : scanner.getIncludedFiles () )
1072        {
1073            final Path file = from.resolve ( relative );
1074            final String targetName = makeUnix ( targetPrefix + relative );
1075
1076            if ( java.nio.file.Files.isSymbolicLink ( file ) )
1077            {
1078                RpmMojo.this.logger.debug ( "%s%s (symlink)", padding, file );
1079                if ( collector.isSymbolicLinks () )
1080                {
1081                    final Path sym = java.nio.file.Files.readSymbolicLink ( file );
1082                    RpmMojo.this.logger.debug ( "%s%s (symlink)", padding, file );
1083                    RpmMojo.this.logger.debug ( "%s  - target: %s", padding, targetName );
1084                    RpmMojo.this.logger.debug ( "%s  - linkTo: %s", padding, sym.toString () );
1085                }
1086                else
1087                {
1088                    RpmMojo.this.logger.debug ( "%s%s (symlink) - ignoring symbolic links", padding, file );
1089                }
1090            }
1091            else
1092            {
1093                RpmMojo.this.logger.debug ( "%s%s (file)", padding, file );
1094                RpmMojo.this.logger.debug ( "%s  - target: %s", padding, targetName );
1095
1096                ctx.addFile ( targetName, file, provider );
1097            }
1098        }
1099    }

1246    private String makeHostname ()
1247    {
1248        String hostname;
1249
1250        try
1251        {
1252            hostname = readFirstLine ( new File ( "/etc/hostname" ), StandardCharsets.US_ASCII );
1253
1254            if ( hostname != null && !hostname.isEmpty () )
1255            {
1256                this.logger.debug ( "Hostname: from /etc/hostname -> '%s'", hostname );
1257                return hostname;
1258            }
1259        }
1260        catch ( final IOException e )
1261        {
1262        }
1263
1264        hostname = System.getenv ( "COMPUTERNAME" );
1265        if ( hostname != null && !hostname.isEmpty () )
1266        {
1267            this.logger.debug ( "Hostname: from COMPUTERNAME -> '%s'", hostname );
1268            return hostname.toLowerCase ();
1269        }
1270
1271        hostname = System.getenv ( "hostname" );
1272        if ( hostname != null && !hostname.isEmpty () )
1273        {
1274            this.logger.debug ( "Hostname: from hostname -> '%s'", hostname );
1275            return hostname;
1276        }
1277
1278        try
1279        {
1280            hostname = InetAddress.getLocalHost ().getHostName ();
1281            this.logger.debug ( "Hostname: from lookup -> '%s'", hostname );
1282            return hostname;
1283        }
1284        catch ( final UnknownHostException e )
1285        {
1286            this.logger.debug ( "Hostname: Falling back to 'localhost'" );
1287            return "localhost";
1288        }
1289    }
