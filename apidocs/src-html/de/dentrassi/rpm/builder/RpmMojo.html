641    @Override
642    public void execute () throws MojoExecutionException, MojoFailureException
643    {
644        this.logger = new Logger ( getLog () );
645
646        if ( this.skip )
647        {
648            this.logger.debug ( "Skipping execution" );
649            return;
650        }
651
652        this.eval = new RulesetEvaluator ( this.rulesets );
653
654        final Path targetDir;
655
656        if ( this.targetDir != null )
657        {
658            targetDir = this.targetDir.toPath ();
659        }
660        else
661        {
662            targetDir = Paths.get ( this.project.getBuild ().getDirectory () );
663        }
664
665        if ( !Files.exists ( targetDir ) )
666        {
667            try
668            {
669                Files.createDirectories ( targetDir );
670            }
671            catch ( final FileAlreadyExistsException e )
672            {
673                // silently ignore
674            }
675            catch ( final IOException ioe )
676            {
677                this.logger.debug ( "Unable to create target directory {}", targetDir );
678                throw new MojoExecutionException ( "RPM build failed.", ioe );
679
680            }
681        }
682
683        final Path targetFile = makeTargetFile ( targetDir );
684
685        this.logger.debug ( "Max supported RPM version: %s", this.maximumSupportedRpmVersion );
686
687        this.logger.info ( "Writing to target to: %s", targetFile );
688        this.logger.debug ( "Default script interpreter: %s", this.defaultScriptInterpreter );
689        this.logger.debug ( "Default ruleset: %s", this.defaultRuleset );
690
691        final String packageName = makePackageName ();
692        final RpmVersion version = makeVersion ();
693
694        this.logger.info ( "RPM base information - name: %s, version: %s, arch: %s", packageName, version, this.architecture );
695
696        testLeadFlags ();
697
698        try ( final RpmBuilder builder = new RpmBuilder ( packageName, version, this.architecture, targetFile ) )
699        {
700            this.logger.info ( "Writing target file: %s", builder.getTargetFile () );
701
702            if ( this.leadOverrideArchitecture != null )
703            {
704                this.logger.info ( "Override RPM lead architecture: %s", this.leadOverrideArchitecture );
705                builder.setLeadOverrideArchitecture ( this.leadOverrideArchitecture );
706            }
707            if ( this.leadOverrideOperatingSystem != null )
708            {
709                this.logger.info ( "Override RPM lead operating system: %s", this.leadOverrideOperatingSystem );
710                builder.setLeadOverrideOperatingSystem ( this.leadOverrideOperatingSystem );
711            }
712
713            fillPackageInformation ( builder );
714            fillScripts ( builder );
715            fillDependencies ( builder );
716            fillPayload ( builder );
717            fillPrefixes ( builder );
718
719            // add signer
720
721            if ( !this.skipSigning && this.signature != null )
722            {
723                final SignatureProcessor signer = makeRsaSigner ( this.signature );
724                if ( signer != null )
725                {
726                    builder.addSignatureProcessor ( signer );
727                }
728            }
729
730            // finally build the file
731
732            builder.build ();
733
734            // version check
735
736            checkVersion ( builder );
737
738            // attach when necessary
739
740            if ( this.attach )
741            {
742                this.projectHelper.attachArtifact ( this.project, "rpm", this.classifier, builder.getTargetFile ().toFile () );
743            }
744        }
745        catch ( final IOException e )
746        {
747            throw new MojoExecutionException ( "Failed to write RPM", e );
748        }
749    }

751    private Path makeTargetFile ( final Path targetDir )
752    {
753        String outputFileName = this.outputFileName;
754
755        if ( outputFileName == null || outputFileName.isEmpty () )
756        {
757            switch ( this.naming.getDefaultFormat () )
758            {
759                case LEGACY:
760                    outputFileName = RpmFileNameProvider.LEGACY_FILENAME_PROVIDER.getRpmFileName ( makePackageName (), makeVersion (), this.architecture );
761                    break;
762                default:

767        }

770        this.logger.debug ( "Resolved output file name - fileName: %s, fullName: %s", this.outputFileName, targetFile );
771        return targetFile;
772    }

858    private <T extends SimpleDependency> void addAllDependencies ( final String depName, final List<T> deps, final DependencyAdder adder, final Consumer<T> validator, final Consumer<Set<RpmDependencyFlags>> flagsCustomizer )
859    {

863        }
864
865        for ( final T dep : deps )

870            final String version = dep.getVersion ();
871            final Set<RpmDependencyFlags> flags = dep.getFlags ();
872
873            if ( flagsCustomizer != null )

875                flagsCustomizer.accept ( flags );
876            }
877
878            this.logger.info ( "Adding dependency [%s]: name = %s, version = %s, flags = %s", depName, name, version, flags );
879
880            adder.add ( name, version, flags.toArray ( new RpmDependencyFlags[0] ) );

892    private void setScript ( final String scriptName, final Script script, final ScriptSetter setter ) throws IOException
893    {
894        if ( script == null )
895        {
896            return;
897        }
898
899        final String scriptContent = script.makeScriptContent ();
900
901        if ( Strings.isNullOrEmpty ( scriptContent ) )
902        {
903            return;
904        }
905
906        String interpreter = script.getInterpreter ();
907        this.logger.debug ( "[script %s:]: explicit interpreter: %s", scriptName, interpreter );
908
909        if ( Strings.isNullOrEmpty ( interpreter ) )
910        {
911            interpreter = detectInterpreter ( scriptContent );
912            this.logger.debug ( "[script %s:]: detected interpreter: %s", scriptName, interpreter );
913        }
914        if ( Strings.isNullOrEmpty ( interpreter ) )
915        {
916            interpreter = this.defaultScriptInterpreter;
917            this.logger.debug ( "[script %s:]: default interpreter: %s", scriptName, interpreter );
918        }
919        this.logger.info ( "[script %s]: Using script interpreter: %s", scriptName, interpreter );
920        this.logger.debug ( "[script %s]: %s", scriptName, scriptContent );
921
922        setter.accept ( interpreter, scriptContent );
923    }

925    private String detectInterpreter ( final String scriptContent ) throws IOException
926    {
927        final String firstLine = CharSource.wrap ( scriptContent ).readFirstLine ();
928        if ( Strings.isNullOrEmpty ( firstLine ) )
929        {
930            return null;
931        }
932
933        if ( firstLine.startsWith ( "#!" ) && firstLine.length () > 2 )
934        {
935            return firstLine.substring ( 2 );
936        }
937
938        return null;
939    }

1027    private void fillFromEntryCollect ( final BuilderContext ctx, final PackageEntry entry ) throws IOException
1028    {
1029        this.logger.debug ( "    as collector:" );
1030
1031        final Collector collector = entry.getCollect ();
1032
1033        this.logger.debug ( "      - configuration: %s", collector );
1034
1035        final String padding = "          ";
1036
1037        final Path from = collector.getFrom ().toPath ();
1038        final String targetPrefix = entry.getName ().endsWith ( "/" ) ? entry.getName () : entry.getName () + "/";
1039
1040        this.logger.debug ( "      - files:" );
1041
1042        final MojoFileInformationProvider provider = makeProvider ( entry, "            - " );
1043
1044        final DirectoryScanner scanner = new DirectoryScanner ();
1045        scanner.setBasedir ( from.toFile () );
1046        scanner.setCaseSensitive ( true );
1047        scanner.setFollowSymlinks ( true );
1048        scanner.setIncludes ( collector.getIncludes () );
1049        scanner.setExcludes ( collector.getExcludes () );
1050        scanner.scan ();
1051
1052        if ( collector.isDirectories () )
1053        {
1054            for ( final String directory : scanner.getIncludedDirectories () )
1055            {
1056                final Path dir = from.resolve ( directory );
1057                if ( dir.equals ( from ) )
1058                {
1059                    continue;
1060                }
1061
1062                RpmMojo.this.logger.debug ( "%s%s (dir)", padding, dir );
1063                final Path relative = from.relativize ( dir );
1064                final String targetName = makeUnix ( targetPrefix + relative.toString () );
1065                RpmMojo.this.logger.debug ( "%s  - target: %s", padding, targetName );
1066                ctx.addDirectory ( targetName, provider );
1067            }
1068        }
1069
1070        for ( final String relative : scanner.getIncludedFiles () )
1071        {
1072            final Path file = from.resolve ( relative );
1073            final String targetName = makeUnix ( targetPrefix + relative );
1074
1075            if ( java.nio.file.Files.isSymbolicLink ( file ) )
1076            {
1077                RpmMojo.this.logger.debug ( "%s%s (symlink)", padding, file );
1078                if ( collector.isSymbolicLinks () )
1079                {
1080                    final Path sym = java.nio.file.Files.readSymbolicLink ( file );
1081                    RpmMojo.this.logger.debug ( "%s%s (symlink)", padding, file );
1082                    RpmMojo.this.logger.debug ( "%s  - target: %s", padding, targetName );
1083                    RpmMojo.this.logger.debug ( "%s  - linkTo: %s", padding, sym.toString () );
1084                }
1085                else
1086                {
1087                    RpmMojo.this.logger.debug ( "%s%s (symlink) - ignoring symbolic links", padding, file );
1088                }
1089            }
1090            else
1091            {
1092                RpmMojo.this.logger.debug ( "%s%s (file)", padding, file );
1093                RpmMojo.this.logger.debug ( "%s  - target: %s", padding, targetName );
1094
1095                ctx.addFile ( targetName, file, provider );
1096            }
1097        }
1098    }

1245    private String makeHostname ()
1246    {
1247        String hostname;
1248
1249        try
1250        {
1251            hostname = readFirstLine ( new File ( "/etc/hostname" ), StandardCharsets.US_ASCII );
1252
1253            if ( hostname != null && !hostname.isEmpty () )
1254            {
1255                this.logger.debug ( "Hostname: from /etc/hostname -> '%s'", hostname );
1256                return hostname;
1257            }
1258        }
1259        catch ( final IOException e )
1260        {
1261        }
1262
1263        hostname = System.getenv ( "COMPUTERNAME" );
1264        if ( hostname != null && !hostname.isEmpty () )
1265        {
1266            this.logger.debug ( "Hostname: from COMPUTERNAME -> '%s'", hostname );
1267            return hostname.toLowerCase ();
1268        }
1269
1270        hostname = System.getenv ( "hostname" );
1271        if ( hostname != null && !hostname.isEmpty () )
1272        {
1273            this.logger.debug ( "Hostname: from hostname -> '%s'", hostname );
1274            return hostname;
1275        }
1276
1277        try
1278        {
1279            hostname = InetAddress.getLocalHost ().getHostName ();
1280            this.logger.debug ( "Hostname: from lookup -> '%s'", hostname );
1281            return hostname;
1282        }
1283        catch ( final UnknownHostException e )
1284        {
1285            this.logger.debug ( "Hostname: Falling back to 'localhost'" );
1286            return "localhost";
1287        }
1288    }
