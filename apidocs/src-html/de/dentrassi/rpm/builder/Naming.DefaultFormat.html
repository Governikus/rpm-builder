011package de.dentrassi.rpm.builder;
012
013public class Naming
014{
015    public static enum Case
016    {
017        UNMODIFIED,
018        LOWERCASE
019    }
020
021    public static enum DefaultFormat
022    {
023        DEFAULT,
024        LEGACY;
025    }
026
027    private Case caseValue = Case.UNMODIFIED;
028
029    private DefaultFormat defaultFormat = DefaultFormat.DEFAULT;
030
031    public void setCase ( final String caseValue )
032    {
033        if ( caseValue != null )
034        {
035            this.caseValue = Case.valueOf ( caseValue.toUpperCase () );
036        }
037        else
038        {
039            this.caseValue = Case.UNMODIFIED;
040        }
041    }
042
043    public Case getCase ()
044    {
045        return this.caseValue;
046    }
047
048    public void setDefaultFormat ( final String defaultFormat )
049    {
050        if ( defaultFormat != null )
051        {
052            this.defaultFormat = DefaultFormat.valueOf ( defaultFormat.toUpperCase () );
053        }
054        else
055        {
056            this.defaultFormat = DefaultFormat.DEFAULT;
057        }
058    }
059
060    public DefaultFormat getDefaultFormat ()
061    {
062        return this.defaultFormat;
063    }
064}
