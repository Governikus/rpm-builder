011package de.dentrassi.rpm.builder;
012
013import java.util.Set;
014
015import org.eclipse.packagedrone.utils.rpm.deps.RpmDependencyFlags;
016
017import com.google.common.base.Strings;
018
019public class Dependency extends SimpleDependency
020{
021    public void setName ( final String name )
022    {
023        this.name = name;
024    }
025
026    public void setVersion ( final String version )
027    {
028        this.version = version;
029    }
030
031    public void setFlags ( final Set<RpmDependencyFlags> flags )
032    {
033        this.flags.addAll ( flags );
034    }
035
036    public void setGe ( final Object ignored )
037    {
038        setGreaterOrEqual ( null );
039    }
040
041    public void setGreaterOrEqual ( final Object ignored )
042    {
043        this.flags.add ( RpmDependencyFlags.GREATER );
044        this.flags.add ( RpmDependencyFlags.EQUAL );
045    }
046
047    public void setLe ( final Object ignored )
048    {
049        setLessOrEqual ( null );
050    }
051
052    public void setLessOrEqual ( final Object ignored )
053    {
054        this.flags.add ( RpmDependencyFlags.LESS );
055        this.flags.add ( RpmDependencyFlags.EQUAL );
056    }
057
058    public void setEq ( final Object ignored )
059    {
060        this.flags.add ( RpmDependencyFlags.EQUAL );
061    }
062
063    public void setEqual ( final Object ignored )
064    {
065        this.flags.add ( RpmDependencyFlags.EQUAL );
066    }
067
068    public void setGreater ( final Object ignored )
069    {
070        this.flags.add ( RpmDependencyFlags.GREATER );
071    }
072
073    public void setLess ( final Object ignored )
074    {
075        this.flags.add ( RpmDependencyFlags.LESS );
076    }
077
078    public void setPre ( final Object ignored )
079    {
080        this.flags.add ( RpmDependencyFlags.PREREQ );
081    }
082
083    public void validate ()
084    {
085        if ( Strings.isNullOrEmpty ( this.name ) )
086        {
087            throw new IllegalStateException ( "'name' of dependency must be set" );
088        }
089    }
090}
