011package de.dentrassi.rpm.builder;
012
013import java.io.FileInputStream;
014import java.io.IOException;
015import java.io.InputStream;
016
017import org.apache.maven.plugin.MojoExecutionException;
018import org.apache.maven.plugin.MojoFailureException;
019import org.bouncycastle.openpgp.PGPException;
020import org.bouncycastle.openpgp.PGPPrivateKey;
021import org.eclipse.packagedrone.utils.security.pgp.PgpHelper;
022
023public final class SigningHelper
024{
025
026    private SigningHelper ()
027    {
028    }
029
030    public static PGPPrivateKey loadKey ( final Signature signature, final Logger logger ) throws MojoFailureException, MojoExecutionException
031    {
032        if ( signature == null )
033        {
034            return null;
035        }
036
037        if ( signature.isSkip () )
038        {
039            return null;
040        }
041
042        if ( signature.getKeyId () == null || signature.getKeyId ().isEmpty () )
043        {
044            throw new MojoFailureException ( signature, "'keyId' parameter not set", "Signing requires the 'keyId' to the user id of the GPG key to use." );
045        }
046
047        if ( signature.getKeyringFile () == null )
048        {
049            throw new MojoFailureException ( signature, "'keyringFile' parameter not set", "Signing requires the 'keyringFile' to be set to a valid GPG keyring file, containing the secret keys." );
050        }
051
052        if ( signature.getPassphrase () == null )
053        {
054            throw new MojoFailureException ( signature, "'passphrase' parameter not set", "Signing requires the 'passphrase' parameter to be set." );
055        }
056
057        try ( InputStream input = new FileInputStream ( signature.getKeyringFile () ) )
058        {
059            final PGPPrivateKey privateKey = PgpHelper.loadPrivateKey ( input, signature.getKeyId (), signature.getPassphrase () );
060            if ( privateKey == null )
061            {
062                throw new MojoFailureException ( String.format ( "Unable to load GPG key '%s' from '%s'", signature.getKeyId (), signature.getKeyringFile () ) );
063            }
064            logger.info ( "Signing RPM - keyId: %016x", privateKey.getKeyID () );
065            return privateKey;
066        }
067        catch ( final PGPException | IOException e )
068        {
069            throw new MojoExecutionException ( "Failed to load private key for signing", e );
070        }
071    }
072}
