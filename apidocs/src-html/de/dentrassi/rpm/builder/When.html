012package de.dentrassi.rpm.builder;
013
014import org.eclipse.packagedrone.utils.rpm.build.PayloadEntryType;
015import org.slf4j.LoggerFactory;
016
017public class When
018{
019    private static final org.slf4j.Logger logger = LoggerFactory.getLogger ( When.class );
020
021    private String type;
022
023    private String prefix;
024
025    private String suffix;
026
027    public void setType ( final String type )
028    {
029        this.type = type;
030    }
031
032    public String getType ()
033    {
034        return this.type;
035    }
036
037    public void setPrefix ( final String prefix )
038    {
039        this.prefix = prefix;
040    }
041
042    public String getPrefix ()
043    {
044        return this.prefix;
045    }
046
047    public void setSuffix ( final String suffix )
048    {
049        this.suffix = suffix;
050    }
051
052    public String getSuffix ()
053    {
054        return this.suffix;
055    }
056
057    @Override
058    public String toString ()
059    {
060        return String.format ( "[when - type: %s, prefix: %s, suffix: %s]", this.type, this.prefix, this.suffix );
061    }
062
063    public boolean matches ( final Object object, final PayloadEntryType type, final String targetName )
064    {
065        if ( this.prefix != null && !this.prefix.isEmpty () && !targetName.startsWith ( this.prefix ) )
066        {
067            logger.debug ( "Prefix is set and does not match - expected: '{}', provided: '{}'", this.prefix, targetName );
068            return false;
069        }
070
071        if ( this.suffix != null && !this.suffix.isEmpty () && !targetName.endsWith ( this.suffix ) )
072        {
073            logger.debug ( "Suffix is set and does not match - expected: '{}', provided: '{}'", this.suffix, targetName );
074            return false;
075        }
076
077        if ( this.type != null && !this.type.isEmpty () )
078        {
079            logger.debug ( "Testing type - expected: {}, actual: {}", this.type, type );
080
081            switch ( this.type.toLowerCase () )
082            {
083                case "directory":
084                    if ( type != PayloadEntryType.DIRECTORY )
085                    {
086                        return false;
087                    }
088                    break;
089                case "file":
090                    if ( type != PayloadEntryType.FILE )
091                    {
092                        return false;
093                    }
094                    break;
095                case "link":
096                    if ( type != PayloadEntryType.SYMBOLIC_LINK )
097                    {
098                        return false;
099                    }
100                    break;
101                default:
102                    throw new IllegalStateException ( String.format ( "Unknown match type: '%s'", this.type ) );
103            }
104        }
105
106        logger.debug ( "Is a match" );
107
108        return true;
109    }
110}
